function populateBigBoard() { const bigBoard = JSON.parse(localStorage.getItem("bigBoard")) || []; const boardDiv = document.getElementById("big-board"); if (bigBoard.length === 0) { boardDiv.innerHTML = "

No players on the Big Board yet.

"; return; } boardDiv.innerHTML = bigBoard .map(player => `

${player.name}

College: ${player.college}

Position: ${player.position}

Skills:

    ${Object.entries(player.skills) .map(([skill, rating]) => `
  • ${skill}: ${rating}
    • `) .join("")}
`) .join(""); }
